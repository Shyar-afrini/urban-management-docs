# POST /auth/signin

Sign in a user with email and password. Returns a session token you can use for authenticated requests.

```mermaid
sequenceDiagram
    participant Client
    participant Server

    Client->>Server: Send credentials (username & password)

    Server-->>Client: { "success": true }
    Note over Server,Client: Set-Cookie: session_token
```

export async function signIn(email: string, password: string): Promise<SignInResponse> {
  try {
    const { data } = await api.post<SignInResponse>("/auth/signin", { email, password });
    return data;
  } catch (err) {
    return normalizeAxiosError(err);
  }
}
