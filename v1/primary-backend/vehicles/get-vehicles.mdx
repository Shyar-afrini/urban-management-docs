function parse(
  schema: ZodSchema<selectVehicleSchema>,
  data: unknown
): TResult<Vehicle> {
  const result = schema.safeParse(data);
  if (!result.success) {
    return {
      success: false,
      error: {
        code: BAD_REQUEST_STATUS,
        message: BAD_REQUEST_ERR,
        details: result.error.flatten(),
      },
    };
  }
  return { success: true, data: result.data };
}
